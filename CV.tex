\section*{Working Paper}

\begin{tabular*}{1.0\textwidth}[t]{l@{\extracolsep{\fill}}r}
  \textbf{Institutional Common Ownership and Executive Mobility} & \\
  \textit{Supervisor: Prof.\ Jos\'e Azar (IESE \& UNAV)} & \\
\end{tabular*}\vspace{-5pt}
\begin{itemize}[leftmargin=0.15in, label={--}]
    \resumeItem{Exploiting the BlackRock--BGI merger (2009) as a quasi-natural experiment, I examine how institutional common ownership shapes C-suite executive mobility.}
    \resumeItem{Using difference-in-differences and event study designs on a 27.8-million-row firm-pair panel linking 13F holdings (58.9M obs.), ExecuComp data (31,959 executive-years), and Orbis director moves (24,716 transitions), I find that common ownership increases mobility through within-network labor market channels---rejecting the shadow non-compete hypothesis.}
    \resumeItem{\textbf{Results:} Firm-level DiD: +0.073 ($p = 0.002$). Pair-level logit: $\Delta\lambda = +0.582$ ($p < 10^{-18}$). Event study: Post-2009 significant ($p < 0.001$). Robust to 17 pair-level checks, 5 placebo tests.}
    \resumeItem{\textbf{Link:} \href{https://github.com/kunkunz420/common-ownership-executive-mobility}{github.com/kunkunz420/common-ownership-executive-mobility}}
\end{itemize}
